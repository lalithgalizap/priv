End-to-End Production Technical SpecificationAnonymous AI Mediation Platform (Supabase Auth Edition)
1. Executive Summary & Product Requirements
This Product Requirements Document (PRD) outlines the framework for an AI mediation application tailored for medium-scale businesses. The core value proposition is total data anonymity: serving as a secure broker that scrubs corporate footprints and Personally Identifiable Information (PII) before requests are routed to cloud-hosted upstream AI providers.
Core Objectives
Anonymity as a Service: Zero trace of user identity or enterprise metadata sent to final LLMs.
Zero-Cost Bootstrapping: Maximize infrastructure allocation using cloud free tiers up to 50k MAU.
Premium Enterprise UI/UX: Dark-theme default workspace designed to look completely premium.
Isolated Multi-Tenancy: Relational grouping of users by company without merging data lines.

3. Authentication Configuration (Supabase Auth)
To replace AWS Cognito while maintaining a 50,000 MAU free tier, the platform will utilize Supabase Auth. Since Supabase is already utilized for the PostgreSQL database, this consolidates vendors and simplifies JWT validation. The Python backend will verify the HS256 JWT signature using the Supabase project secret.
import jwtfrom fastapi import HTTPException, Security, statusfrom fastapi.security import HTTPBearer, HTTPAuthorizationCredentialsfrom pydantic import BaseModel# Supabase Project JWT Secret (Found in API Settings)SUPABASE_JWT_SECRET = "your-super-secret-jwt-token-with-at-least-32-characters-long"class UserSession(BaseModel):    user_id: str    email: str    tenant_id: strsecurity_backend = HTTPBearer()def get_current_user(credentials: HTTPAuthorizationCredentials = Security(security_backend)) -> UserSession:    try:        token = credentials.credentials                # Supabase uses HS256 to sign its JWTs        payload = jwt.decode(            token,             SUPABASE_JWT_SECRET,             algorithms=["HS256"],             audience="authenticated"        )                # Extract user identity and custom tenant mapping from JWT user_metadata        return UserSession(            user_id=payload.get("sub"),            email=payload.get("email", ""),            tenant_id=payload.get("user_metadata", {}).get("tenant_id", "default_tenant")        )    except jwt.ExpiredSignatureError:        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token has expired.")    except jwt.PyJWTError:        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid authentication credentials.")

import timeimport requestsfrom fastapi import FastAPI, Depends, HTTPException, statusfrom pydantic import BaseModel, Fieldimport psycopg2from app.middleware.auth import get_current_user, UserSessionfrom app.middleware.anonymizer import anonymizer_engineapp = FastAPI(title="Core Mediation Server Platform")DATABASE_URL = "postgresql://postgres:your_password@db.supabase.co:5432/postgres"class MediationPayload(BaseModel):    prompt: str = Field(..., min_length=1, max_length=10000)    model: str = Field(default="gpt-4")@app.post("/api/v1/mediate")async def execute_anonymous_brokerage(payload: MediationPayload, current_user: UserSession = Depends(get_current_user)):    start_time = time.time()        # Text Redaction Sequence Engine    anonymized_text = anonymizer_engine.sanitize_prompt(payload.prompt)        # Transmission Map Setup    upstream_headers = {"Authorization": "Bearer MASTER_KEY", "Content-Type": "application/json"}    upstream_payload = {        "model": payload.model,        "messages": [{"role": "user", "content": anonymized_text}]    }        try:        response = requests.post("https://api.openai.com/v1/chat/completions", json=upstream_payload, headers=upstream_headers, timeout=45)        res_json = response.json()                in_tokens = res_json["usage"]["prompt_tokens"]        out_tokens = res_json["usage"]["completion_tokens"]        tot_tokens = res_json["usage"]["total_tokens"]        duration = int((time.time() - start_time) * 1000)                # Telemetry Entry Tracking (Saves zero plain text strings)        conn = psycopg2.connect(DATABASE_URL)        cur = conn.cursor()        cur.execute(            "INSERT INTO tenant_usage_metrics (tenant_id, supabase_auth_id, model_identifier, input_tokens_used, output_tokens_used, total_tokens_used, execution_duration_ms) VALUES (%s, %s, %s, %s, %s, %s, %s)",            (current_user.tenant_id, current_user.user_id, payload.model, in_tokens, out_tokens, tot_tokens, duration)        )        conn.commit()        cur.close()        conn.close()                return {            "sanitized_prompt_preview": anonymized_text[:100],            "ai_response": res_json["choices"][0]["message"]["content"],            "tokens_processed": tot_tokens        }    except Exception as e:        raise HTTPException(status_code=500, detail="Broker pipeline transaction dropped.")
